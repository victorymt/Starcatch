tui/src/main.rs — 入口

fn main() -> io::Result<()> {
    let db_path = /* 参数或默认 ~/.local/share/starcatch/starcatch.db */;
    enable_raw_mode();
    let mut terminal = Terminal::new(CrosstermBackend::new(stdout));
    let mut app = App::new(&db_path);
    loop {
        terminal.draw(|f| ui::draw(f, &app));
        let evt = event::read()?;
        if handler::handle(&mut app, evt)? { break; }
        if app.needs_refresh { app.refresh_data(); }
    }
}
